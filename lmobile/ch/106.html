Sub WriteData(Desc)
	On Error Resume Next
	Dim oFs,OutFile
	set oFs = server.CreateObject("Scripting.FileSystemObject")
	set OutFile = oFs.OpenTextFile("C:\inetpub\wwwroot\upData.txt", 8, true)
	call OutFile.writeLine(Desc)
	OutFile.close 
end Sub
